-- Hàm xử lý Path nghiêm ngặt - Sai là báo lỗi ngay chứ không fallback bậy
local function safeResolvePath(pathStr)
    if not pathStr or pathStr == "" then return workspace, "OK" end
    
    -- Giữ nguyên ký tự bên trong tên, chỉ lọc khoảng trắng thừa ở đầu/cuối đường dẫn nếu có
    pathStr = string.gsub(pathStr, "^%s*(.-)%s*$", "%1")
    
    local segments = {}
    for segment in string.gmatch(pathStr, "[^.]+") do 
        table.insert(segments, segment) 
    end
    
    if #segments == 0 then return workspace, "OK" end
    
    local current = game
    local startIndex = 1
    
    -- Phân tích gốc ban đầu
    if segments[1] == "workspace" or segments[1] == "Workspace" then
        current = workspace
        startIndex = 2
    elseif segments[1] == "game" then
        if segments[2] == "workspace" or segments[2] == "Workspace" then
            current = workspace
            startIndex = 3
        else
            current = game
            startIndex = 2
        end
    end
    
    -- Duyệt sâu vào các child
    for i = startIndex, #segments do
        local name = segments[i]
        local success, nextObj = pcall(function() 
            return current:FindFirstChild(name) 
        end)
        
        if success and nextObj then 
            current = nextObj 
        else 
            return nil, "❌ Không tìm thấy '" .. name .. "' bên trong '" .. current:GetFullName() .. "'"
        end
    end
    
    return current, "OK"
end

-- TÍNH NĂNG SAVE BUILD TRÚNG ĐÍCH
local function serializeInstanceV7(pathString)
    local targetModel, errMsg = safeResolvePath(pathString)
    
    -- Nếu không tìm thấy mục tiêu theo chuỗi nhập vào, dừng ngay và hiện lỗi cho bro sửa
    if not targetModel then 
        if TPHAT_UI_StatusLabel then 
            TPHAT_UI_StatusLabel.Text = errMsg
            TPHAT_UI_StatusLabel.TextColor3 = Color3.fromRGB(255, 100, 100)
        end
        sendNotify("❌ Lỗi Đường Dẫn!", "Vui lòng kiểm tra lại tên Object!")
        return 
    end
    
    if TPHAT_UI_StatusLabel then 
        TPHAT_UI_StatusLabel.Text = "⏳ Đang cào cấu trúc của: " .. targetModel.Name
        TPHAT_UI_StatusLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
    end
    if TPHAT_UI_ProgressBar then TPHAT_UI_ProgressBar.Size = UDim2.new(0, 0, 1, 0) end
    task.wait(0.05)
    
    -- Thu thập chính xác: Chỉ lấy bản thân nó và con cháu từ nhánh đó xuống
    local descendants = {}
    pcall(function() descendants = targetModel:GetDescendants() end)
    table.insert(descendants, 1, targetModel)
    local totalObjects = #descendants
    
    local objectToId = {}
    local stringDict = {}
    local stringCounter = 0
    
    local function getStringId(str)
        if not str or str == "" then return nil end
        if stringDict[str] then return stringDict[str] end
        stringCounter += 1 stringDict[str] = stringCounter
        return stringCounter
    end
    
    for id, obj in ipairs(descendants) do objectToId[obj] = id end
    local mainDataBuffer = table.create(totalObjects)
    local validCount = 0
    
    for i, object in ipairs(descendants) do
        if i % 100 == 0 then task.wait() end
        
        local successProcess, partInfo = pcall(function()
            local className = object.ClassName
            local isBasePart = object:IsA("BasePart")
            
            if isBasePart or object:IsA("Attachment") or object:IsA("JointInstance") or 
               object:IsA("Decal") or object:IsA("Texture") or object:IsA("Light") or 
               object:IsA("SpecialMesh") or object:IsA("Model") or object:IsA("Folder") then
                
                local parentObj = object.Parent
                local parentId = 0
                if object ~= targetModel and parentObj then
                    parentId = objectToId[parentObj] or 0
                end
                
                local info = { I = i, P = parentId, C = getStringId(className), N = getStringId(object.Name) }
                
                if isBasePart then
                    pcall(function() info.S = {object.Size.X, object.Size.Y, object.Size.Z} end)
                    pcall(function() info.M = getStringId(object.Material.Name) end)
                    pcall(function() info.Col = {math.floor(object.Color.R * 255), math.floor(object.Color.G * 255), math.floor(object.Color.B * 255)} end)
                    pcall(function() info.A = object.Anchored and 1 or 0 end)
                    pcall(function() info.Cc = object.CanCollide and 1 or 0 end)
                    pcall(function() info.Cf = {object.CFrame:GetComponents()} end)
                    
                    if object:IsA("MeshPart") then
                        pcall(function() info.Mid = getStringId(object.MeshId) end)
                        pcall(function() info.Tid = getStringId(object.TextureId) end)
                    elseif object:IsA("UnionOperation") then
                        pcall(function() info.Aid = getStringId(object.AssetId or object.MeshId or "") end)
                    end
                elseif object:IsA("Attachment") then
                    pcall(function() info.Cf = {object.CFrame:GetComponents()} end)
                elseif object:IsA("JointInstance") then
                    pcall(function()
                        info.P0 = objectToId[object.Part0] or 0
                        info.P1 = objectToId[object.Part1] or 0
                        info.C0 = {object.C0:GetComponents()}
                        info.C1 = {object.C1:GetComponents()}
                    end)
                elseif object:IsA("Decal") then
                    pcall(function() info.Tex = getStringId(object.Texture) end)
                    pcall(function() info.Face = getStringId(object.Face.Name) end)
                elseif object:IsA("Light") then
                    pcall(function()
                        info.Col = {math.floor(object.Color.R * 255), math.floor(object.Color.G * 255), math.floor(object.Color.B * 255)}
                        info.Br = object.Brightness info.Rng = object.Range info.En = object.Enabled and 1 or 0
                    end)
                elseif object:IsA("SpecialMesh") then
                    pcall(function() info.Mid = getStringId(object.MeshId) info.Tid = getStringId(object.TextureId) info.Scl = {object.Scale.X, object.Scale.Y, object.Scale.Z} info.Mt = getStringId(object.MeshType.Name) end)
                end
                return info
            end
        end)
        
        if successProcess and partInfo then validCount += 1 mainDataBuffer[validCount] = partInfo end
        if (i % 100 == 0 or i == totalObjects) and TPHAT_UI_StatusLabel and TPHAT_UI_ProgressBar then
            local percent = math.floor((i / totalObjects) * 100)
            TPHAT_UI_StatusLabel.Text = "⚡ Đang cào dữ liệu: " .. percent .. "% (" .. i .. "/" .. totalObjects .. ")"
            TPHAT_UI_ProgressBar.Size = UDim2.new(i / totalObjects, 0, 1, 0)
        end
    end
    
    if validCount < totalObjects then
        local trimmedBuffer = table.create(validCount) table.move(mainDataBuffer, 1, validCount, 1, trimmedBuffer) mainDataBuffer = trimmedBuffer
    end
    
    local flatStringArray = table.create(stringCounter)
    for str, id in pairs(stringDict) do flatStringArray[id] = str end
    
    local finalPackage = { Dictionary = flatStringArray, Data = mainDataBuffer }
    local _, finalJsonOutput = pcall(HttpService.JSONEncode, HttpService, finalPackage)
    
    local folderName = "tphat_save_instance"
    pcall(function() if makefolder then makefolder(folderName) end end)
    
    local nameClean = targetModel and targetModel.Name or "LeakedMap"
    local primaryPath = folderName .. "/" .. nameClean .. "_V7.Build"
    
    pcall(function()
        if isfile and isfile(primaryPath) then
            local randomId = math.random(1000, 9999)
            primaryPath = folderName .. "/" .. nameClean .. "_" .. randomId .. "_V7.Build"
        end
    end)
    
    local successWrite = pcall(function() writefile(primaryPath, finalJsonOutput) end)
    
    if TPHAT_UI_StatusLabel then 
        TPHAT_UI_StatusLabel.Text = "✅ Đã lưu thành công nhóm: " .. nameClean
        TPHAT_UI_StatusLabel.TextColor3 = Color3.fromRGB(0, 255, 100) 
    end
    sendNotify("✅ Thành Công!", "File đã lưu vào thư mục workspace/" .. folderName)
end

-- TÍNH NĂNG TRÍCH XUẤT ĐỌC FILE VÀ HIỆN KHUNG PREVIEW
local function activatePreviewV7(fileNameStr)
    clearPreview()
    if not fileNameStr or fileNameStr == "" then 
        if TPHAT_UI_StatusLabel then TPHAT_UI_StatusLabel.Text = "❌ Vui lòng chọn file trong danh sách trước!" end
        return 
    end
    
    if TPHAT_UI_StatusLabel then TPHAT_UI_StatusLabel.Text = "⏳ Đang nạp dữ liệu xem trước..." end
    task.wait(0.02)
    
    local fileContent = nil
    pcall(function() if readfile then fileContent = readfile("tphat_save_instance/" .. fileNameStr) end end)
    
    if not fileContent then
        if TPHAT_UI_StatusLabel then TPHAT_UI_StatusLabel.Text = "❌ Thất bại khi đọc file Build!" end
        return
    end
    
    local successDecode, package = pcall(HttpService.JSONDecode, HttpService, fileContent)
    if not successDecode or not package or not package.Data then
        if TPHAT_UI_StatusLabel then TPHAT_UI_StatusLabel.Text = "❌ File không đúng định dạng JSON!" end
        return
    end
    
    currentPackageData = package 
    local rawData = package.Data
    
    local minX, minY, minZ = math.huge, math.huge, math.huge
    local maxX, maxY, maxZ = -math.huge, -math.huge, -math.huge
    local hasParts = false
    
    for _, info in ipairs(rawData) do
        if info.Cf and info.S then
            hasParts = true
            local cf = CFrame.new(unpack(info.Cf))
            local pos = cf.Position
            if pos.X < minX then minX = pos.X end if pos.X > maxX then maxX = pos.X end
            if pos.Y < minY then minY = pos.Y end if pos.Y > maxY then maxY = pos.Y end
            if pos.Z < minZ then minZ = pos.Z end if pos.Z > maxZ then maxZ = pos.Z end
        end
    end
    
    local sizeBox = Vector3.new(10, 10, 10) 
    if hasParts then
        sizeBox = Vector3.new(math.max(maxX - minX, 2), math.max(maxY - minY, 2), math.max(maxZ - minZ, 2))
    end
    
    local character = LocalPlayer.Character
    local rootPart = character and character:FindFirstChild("HumanoidRootPart")
    if rootPart then
        previewCFrame = rootPart.CFrame * CFrame.new(0, 0, - (sizeBox.Z / 2 + 10)) 
    else
        previewCFrame = CFrame.new(0, 50, 0)
    end
    
    currentPreviewPart = Instance.new("Part")
    currentPreviewPart.Name = "TPHAT_PREVIEW_ANCHOR"
    currentPreviewPart.Size = sizeBox
    currentPreviewPart.CFrame = previewCFrame
    currentPreviewPart.Transparency = 0.85 
    currentPreviewPart.Color = Color3.fromRGB(255, 0, 150)
    currentPreviewPart.Material = Enum.Material.ForceField
    currentPreviewPart.Anchored = true
    currentPreviewPart.CanCollide = false
    currentPreviewPart.Parent = workspace
    
    currentSelectionBox = Instance.new("SelectionBox")
    currentSelectionBox.Name = "TPHAT_PREVIEW_BORDER"
    currentSelectionBox.Color3 = Color3.fromRGB(255, 0, 127)
    currentSelectionBox.LineThickness = 0.08
    currentSelectionBox.Adornee = currentPreviewPart
    currentSelectionBox.Parent = currentPreviewPart
    
    if TPHAT_UI_StatusLabel then TPHAT_UI_StatusLabel.Text = "🔮 Đã hiện khung Preview trước mặt bro!" end
end

-- TÍNH NĂNG LOAD BUILD CHÍNH THỨC
local function deserializeBuildV7()
    if not currentPackageData then
        if TPHAT_UI_StatusLabel then TPHAT_UI_StatusLabel.Text = "❌ Hãy bật Preview để nạp dữ liệu trước khi Load!" end
        return
    end
    
    if TPHAT_UI_StatusLabel then TPHAT_UI_StatusLabel.Text = "🏗️ Đang lắp ráp map thật vào vị trí..." end
    task.wait(0.05)
    
    local dict = currentPackageData.Dictionary or {}
    local rawData = currentPackageData.Data
    local totalCount = #rawData
    
    local function getStr(id) return id and dict[id] or "" end
    local idToCreatedInstance = {}
    local parentJobs = table.create(totalCount)
    
    local originalRootCFrame = nil
    for _, info in ipairs(rawData) do
        if info.Cf then
            originalRootCFrame = CFrame.new(unpack(info.Cf))
            break
        end
    end
    if not originalRootCFrame then originalRootCFrame = CFrame.new() end
    
    local targetPivotCFrame = previewCFrame
    
    for idx, info in ipairs(rawData) do
        if idx % 100 == 0 then task.wait() end
        
        pcall(function()
            local cName = getStr(info.C)
            local name = getStr(info.N)
            local obj = nil
            
            if cName == "Part" or cName == "MeshPart" or cName == "UnionOperation" or cName == "Model" or cName == "Folder" then
                obj = Instance.new(cName)
            elseif cName == "Attachment" or cName == "Decal" or cName == "Texture" or cName == "SpecialMesh" then
                obj = Instance.new(cName)
            elseif string.find(cName, "Light") then
                obj = Instance.new(cName)
            elseif string.find(cName, "Weld") or string.find(cName, "Motor6D") then
                obj = Instance.new(cName)
            else
                obj = Instance.new("Folder")
            end
            
            if obj then
                obj.Name = name
                if obj:IsA("BasePart") then
                    if info.S then obj.Size = Vector3.new(info.S[1], info.S[2], info.S[3]) end
                    if info.Cf then 
                        local oldCf = CFrame.new(unpack(info.Cf))
                        local offset = originalRootCFrame:ToObjectSpace(oldCf)
                        obj.CFrame = targetPivotCFrame * offset
                    end
                    if info.M then pcall(function() obj.Material = Enum.Material[getStr(info.M)] end) end
                    if info.Col then obj.Color = Color3.fromRGB(info.Col[1], info.Col[2], info.Col[3]) end
                    obj.Anchored = (info.A == 1)
                    obj.CanCollide = (info.Cc == 1)
                    if obj:IsA("MeshPart") then
                        if info.Mid then pcall(function() obj.MeshId = getStr(info.Mid) end) end
                        if info.Tid then pcall(function() obj.TextureId = getStr(info.Tid) end) end
                    end
                elseif obj:IsA("Attachment") then
                    if info.Cf then 
                        local oldCf = CFrame.new(unpack(info.Cf))
                        local offset = originalRootCFrame:ToObjectSpace(oldCf)
                        obj.CFrame = targetPivotCFrame * offset
                    end
                elseif obj:IsA("Decal") then
                    if info.Tex then obj.Texture = getStr(info.Tex) end
                    if info.Face then pcall(function() obj.Face = Enum.NormalId[getStr(info.Face)] end) end
                elseif obj:IsA("Light") then
                    if info.Col then obj.Color = Color3.fromRGB(info.Col[1], info.Col[2], info.Col[3]) end
                    if info.Br then obj.Brightness = info.Br end
                    if info.Rng then obj.Range = info.Rng end
                    obj.Enabled = (info.En == 1)
                elseif obj:IsA("SpecialMesh") then
                    if info.Mid then obj.MeshId = getStr(info.Mid) end
                    if info.Tid then obj.TextureId = getStr(info.Tid) end
                    if info.Scl then obj.Scale = Vector3.new(info.Scl[1], info.Scl[2], info.Scl[3]) end
                    if info.Mt then pcall(function() obj.MeshType = Enum.MeshType[getStr(info.Mt)] end) end
                elseif obj:IsA("JointInstance") then
                    task.spawn(function()
                        repeat task.wait() until idToCreatedInstance[info.P0] and idToCreatedInstance[info.P1]
                        pcall(function() obj.Part0 = idToCreatedInstance[info.P0] end)
                        pcall(function() obj.Part1 = idToCreatedInstance[info.P1] end)
                    end)
                    if info.C0 then obj.C0 = CFrame.new(unpack(info.C0)) end
                    if info.C1 then obj.C1 = CFrame.new(unpack(info.C1)) end
                end
                
                idToCreatedInstance[info.I] = obj
                table.insert(parentJobs, { Obj = obj, ParentId = info.P })
            end
        end)
        
        if (idx % 50 == 0 or idx == totalCount) and TPHAT_UI_StatusLabel and TPHAT_UI_ProgressBar then
            local percent = math.floor((idx / totalCount) * 100)
            TPHAT_UI_StatusLabel.Text = "🔨 Lắp ráp chi tiết: " .. percent .. "% (" .. idx .. "/" .. totalCount .. ")"
            TPHAT_UI_ProgressBar.Size = UDim2.new(idx / totalCount, 0, 1, 0)
        end
    end
    
    local mapContainer = Instance.new("Model", workspace)
    mapContainer.Name = "TPHAT_LOADED_BUILD"
    
    for _, job in ipairs(parentJobs) do
        pcall(function()
            local parentObj = idToCreatedInstance[job.ParentId]
            job.Obj.Parent = parentObj or mapContainer
        end)
    end
    
    clearPreview() 
    currentPackageData = nil
    
    if TPHAT_UI_StatusLabel then TPHAT_UI_StatusLabel.Text = "✅ Đã dựng Build thành công tại vị trí yêu cầu!" end
    sendNotify("🚀 Hoàn Tất!", "Check Map mới tại Workspace!")
end

local function refreshFolderFiles()
    for _, child in ipairs(TPHAT_FileScrollingFrame:GetChildren()) do
        if child:IsA("TextButton") then child:Destroy() end
    end
    
    local files = {}
    local folderName = "tphat_save_instance"
    
    pcall(function()
        if listfiles then
            local allFiles = listfiles(folderName) or {}
            for _, path in ipairs(allFiles) do
                local lowerPath = string.lower(path)
                if string.find(lowerPath, "%.build$") or string.find(lowerPath, "%.build") then
                    local cleanName = path
                    if string.find(path, "/") then
                        cleanName = string.match(path, "([^/]+)$")
                    elseif string.find(path, "\\") then
                        cleanName = string.match(path, "([^\\]+)$")
                    end
                    
                    if cleanName and cleanName ~= "" then
                        table.insert(files, cleanName)
                    end
                end
            end
        end
    end)
    
    if #files == 0 then 
        SelectedFileLabel.Text = "❌ Không tìm thấy file .Build nào!" 
        SelectedFileLabel.TextColor3 = Color3.fromRGB(255, 100, 100)
    else
        SelectedFileLabel.Text = "📁 Tìm thấy " .. #files .. " file. Chọn file bên dưới:"
        SelectedFileLabel.TextColor3 = Color3.fromRGB(180, 180, 180)
    end
    
    TPHAT_FileScrollingFrame.CanvasSize = UDim2.new(0, 0, 0, #files * 28)
    for idx, fName in ipairs(files) do
        local FBtn = Instance.new("TextButton")
        FBtn.Size = UDim2.new(1, -6, 0, 24)
        FBtn.BackgroundColor3 = Color3.fromRGB(33, 33, 33)
        FBtn.Text = " 📄 " .. fName
        FBtn.TextColor3 = Color3.fromRGB(220, 220, 220)
        FBtn.TextSize = 11
        FBtn.TextXAlignment = Enum.TextXAlignment.Left
        FBtn.Parent = TPHAT_FileScrollingFrame
        local UICorner_F = Instance.new("UICorner") UICorner_F.CornerRadius = UDim.new(0, 4) UICorner_F.Parent = FBtn
        FBtn.MouseButton1Click:Connect(function()
            selectedFileName = fName
            SelectedFileLabel.Text = "🎯 Chọn: " .. fName
            SelectedFileLabel.TextColor3 = Color3.fromRGB(0, 180, 255)
        end)
    end
end

RefreshBtn.MouseButton1Click:Connect(refreshFolderFiles)

end)

-- KÉO THẢ GIAO DIỆN
local function makeDraggable(obj, isLogo)
    local dragging, dragStart, startPos
    obj.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            if isLogo or (input.Position.Y - obj.AbsolutePosition.Y < 40) then
                dragging = true dragStart = input.Position startPos = obj.Position
                input.Changed:Connect(function() if input.UserInputState == Enum.UserInputState.End then dragging = false end end)
            end
        end
    end)
    UserInputService.InputChanged:Connect(function(input)
        if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
            local delta = input.Position - dragStart
            obj.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
        end
    end)
end
